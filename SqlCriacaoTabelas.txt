CRIACAO DE TABELAS DO BANCO DE DADOS
////////
CREATE TABLE TB_USUARIO(
  PK_ID_USUARIO BIGINT AUTO_INCREMENT PRIMARY KEY,
  NM_NOME VARCHAR(50),
  NM_SENHA VARCHAR(20),
  NM_CPF VARCHAR(11),
  NM_EMAIL VARCHAR(50),
  DT_DATA_NASCIMENTO DATE  
);
//////
CREATE TABLE TB_CONTA(
   PK_ID_CONTA BIGINT AUTO_INCREMENT PRIMARY KEY,
   FK_ID_USUARIO BIGINT,
   NR_SALDO DECIMAL(18,2),
   NR_AGENCIA INT,
   NR_NUMERO_CONTA INT,
   NR_LIMITE_CREDITO DECIMAL(18,2),
   FOREIGN KEY (FK_ID_USUARIO) REFERENCES TB_USUARIO(PK_ID_USUARIO)
);
///////////
CREATE TABLE TB_TRANSFERENCIA(
  PK_ID_TRANSFERENCIA BIGINT AUTO_INCREMENT PRIMARY KEY,
  FK_ID_CONTA_ORIGEM BIGINT,
  FK_ID_CONTA_DESTINO BIGINT,
  NR_VAOR DECIMAL(18,2),
  DT_DATA DATE,
  TIPO VARCHAR(10),
  FOREIGN KEY (FK_ID_CONTA_ORIGEM) REFERENCES TB_CONTA(PK_ID_CONTA),
  FOREIGN KEY (FK_ID_CONTA_DESTINO) REFERENCES TB_CONTA(PK_ID_CONTA)
);
